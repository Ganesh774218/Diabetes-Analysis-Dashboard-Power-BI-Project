Dax Queries and Measures

BMI Category = 
IF(
    'diabetes new'[BMI] < 18.5, "Underweight", 
    IF(
        'diabetes new'[BMI] <= 24.9, "Normal", 
        IF(
            'diabetes new'[BMI] <= 29.9, "Overweight", 
            "Obese"
        )
    )
)




Diabetes Risk = 
IF(
    'diabetes new'[Glucose] < 100, "Low", 
    IF(
        'diabetes new'[Glucose] <= 125, "Medium", 
        "High"
    )
)



Age Category = IF('diabetes new'[Age]<30,"Less Than 30",
    IF('diabetes new'[Age]>=30 && 'diabetes new'[Age]<=50,"30-50",
    "High Age"
))



PatientCounts = COUNTROWS('diabetes new')


Scaled Target = ('diabetes new'[DiabetesRisk %]/200)*100


DiabetesRisk % = 
VAR totalPatients=COUNTROWS('diabetes new')
//diabetes new is a table name
VAR highRiskPatient=COUNTROWS(FILTER('diabetes new', 'diabetes new'[Glucose]>140 && 'diabetes new'[BMI]>=30
)
)
RETURN
DIVIDE(highRiskPatient,totalPatients,0)*100
